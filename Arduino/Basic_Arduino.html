ðŸ”¦ Example: Blink an LED

int ledPin = 13;

void setup() {
  pinMode(ledPin, OUTPUT);
}

void loop() {
  digitalWrite(ledPin, HIGH); // Turn ON
  delay(1000); // Wait 1 second
  digitalWrite(ledPin, LOW); // Turn OFF
  delay(1000);
}
